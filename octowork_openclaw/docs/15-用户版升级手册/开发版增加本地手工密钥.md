## 一、概述

本文档描述如何在**开发版代码**中新增 License 校验功能，使得：

1. 程序启动时自动校验 `license.key`
2. 未授权 → 显示机器指纹 + 拒绝启动
3. 指纹不匹配 → 拒绝启动（防复制）
4. License 过期 → 拒绝启动
5. 开发者可通过命令行工具快速生成 License

校验逻辑在构建时会被 bytenode 编译为字节码，用户看不到密钥和校验细节。

## 三、Step 1 — 创建机器指纹模块

**文件**: `backend/src/utils/fingerprint.js`

```javascript
/**
 * 机器指纹生成器
 * 
 * 采集 6 个硬件维度做 SHA-256 哈希：
 * 1. hostname   — 电脑名
 * 2. platform   — 操作系统 (darwin/win32/linux)
 * 3. arch       — CPU 架构 (x64/arm64)
 * 4. cpuModel   — CPU 型号
 * 5. totalMem   — 内存总量
 * 6. macAddress  — 第一个非内部网卡的 MAC 地址
 *
 * 同一台电脑每次结果一致；换台电脑就完全不同。
 */
const os = require('os');
const crypto = require('crypto');

function getFirstMacAddress() {
  const interfaces = os.networkInterfaces();
  for (const [name, ifaceList] of Object.entries(interfaces)) {
    for (const iface of ifaceList) {
      if (!iface.internal && iface.mac && iface.mac !== '00:00:00:00:00:00') {
        return iface.mac;
      }
    }
  }
  return 'no-mac';
}

function generateFingerprint() {
  const raw = [
    os.hostname(),
    os.platform(),
    os.arch(),
    os.cpus()[0]?.model || 'unknown-cpu',
    String(os.totalmem()),
    getFirstMacAddress(),
  ].join('|');

  return crypto.createHash('sha256').update(raw).digest('hex').substring(0, 32);
}

module.exports = { generateFingerprint, getFirstMacAddress };
```

## 五、Step 3 — 创建 License 校验器

**文件**: `backend/src/license/verifier.js`

```javascript
/**
 * License 校验器 — 启动时调用
 * 
 * 校验流程:
 *   1. 检查开发模式环境变量（跳过校验）
 *   2. 检查 license.key 是否存在
 *   3. 解码 Base64 → JSON
 *   4. 验证 HMAC-SHA256 签名（防伪造）
 *   5. 验证机器指纹（防复制）
 *   6. 验证过期时间
 * 
 * 任何一步失败 → console.error + process.exit(1)
 */
const crypto = require('crypto');
const fs = require('fs');
const path = require('path');
const { generateFingerprint } = require('../utils/fingerprint');
const { LICENSE_SECRET, LICENSE_PATH, SKIP_LICENSE_ENV } = require('./constants');

function verifyLicense() {
  // ========== 0. 开发模式跳过 ==========
  if (process.env[SKIP_LICENSE_ENV] === '1') {
    console.log('⚙️  开发模式 — 跳过 License 校验');
    return { customer: 'Developer', plan: 'dev', expire: 'unlimited' };
  }

  // ========== 1. 检查 license.key 存在 ==========
  if (!fs.existsSync(LICENSE_PATH)) {
    const fingerprint = generateFingerprint();
    console.error('');
    console.error('╔═══════════════════════════════════════════════════════════╗');
    console.error('║            OctoWork 聊天管理器 — 未激活                    ║');
    console.error('╠═══════════════════════════════════════════════════════════╣');
    console.error('║                                                           ║');
    console.error('║  未检测到授权文件 (license.key)                             ║');
    console.error('║                                                           ║');
    console.error(`║  您的机器指纹: ${fingerprint}  ║`);
    console.error('║                                                           ║');
    console.error('║  请将以上指纹发送给管理员获取授权文件                        ║');
    console.error('║  授权文件放置路径: ~/octowork-chat/license.key              ║');
    console.error('║                                                           ║');
    console.error('╚═══════════════════════════════════════════════════════════╝');
    console.error('');
    process.exit(1);
  }

  // ========== 2. 读取 + Base64 解码 ==========
  let payload;
  try {
    const encoded = fs.readFileSync(LICENSE_PATH, 'utf8').trim();
    const decoded = Buffer.from(encoded, 'base64').toString('utf8');
    payload = JSON.parse(decoded);
  } catch (e) {
    console.error('❌ 授权文件格式错误，无法解析');
    console.error('   请确认 license.key 内容完整且未被修改');
    process.exit(1);
  }

  // ========== 3. 验证 HMAC-SHA256 签名 ==========
  const { signature, ...data } = payload;
  if (!signature) {
    console.error('❌ 授权文件缺少签名字段');
    process.exit(1);
  }

  // 签名时 key 排序保证一致性
  const sortedKeys = Object.keys(data).sort();
  const payloadStr = JSON.stringify(data, sortedKeys);
  const expectedSig = crypto
    .createHmac('sha256', LICENSE_SECRET)
    .update(payloadStr)
    .digest('hex');

  if (signature !== expectedSig) {
    console.error('❌ 授权文件签名无效（文件可能被篡改）');
    process.exit(1);
  }

  // ========== 4. 验证机器指纹 ==========
  const currentFingerprint = generateFingerprint();
  if (payload.fingerprint !== currentFingerprint) {
    console.error('');
    console.error('╔═══════════════════════════════════════════════════════════╗');
    console.error('║           OctoWork 聊天管理器 — 授权设备不匹配             ║');
    console.error('╠═══════════════════════════════════════════════════════════╣');
    console.error('║                                                           ║');
    console.error('║  此授权仅限原设备使用，不可复制到其他电脑                    ║');
    console.error('║                                                           ║');
    console.error(`║  授权设备指纹: ${payload.fingerprint}  ║`);
    console.error(`║  当前设备指纹: ${currentFingerprint}  ║`);
    console.error('║                                                           ║');
    console.error('║  如需迁移到新设备，请联系管理员重新授权                      ║');
    console.error('║                                                           ║');
    console.error('╚═══════════════════════════════════════════════════════════╝');
    console.error('');
    process.exit(1);
  }

  // ========== 5. 验证过期时间 ==========
  if (payload.expire && payload.expire !== 'unlimited') {
    const expireDate = new Date(payload.expire + 'T23:59:59');
    if (isNaN(expireDate.getTime())) {
      console.error('❌ 授权文件中的过期日期格式错误');
      process.exit(1);
    }
    if (Date.now() > expireDate.getTime()) {
      console.error('');
      console.error(`❌ 授权已过期 (${payload.expire})`);
      console.error('   请联系管理员获取新的授权文件');
      console.error('');
      process.exit(1);
    }

    // 提醒即将过期（30 天内）
    const daysLeft = Math.ceil((expireDate.getTime() - Date.now()) / (1000 * 60 * 60 * 24));
    if (daysLeft <= 30) {
      console.warn(`⚠️  授权将在 ${daysLeft} 天后过期 (${payload.expire})，请提前联系管理员续期`);
    }
  }

  // ========== 6. 校验通过 ==========
  console.log(`🔑 授权校验通过 — ${payload.customer}${payload.company ? ' (' + payload.company + ')' : ''} | ${payload.plan} | 有效期至 ${payload.expire}`);

  return payload;
}

/**
 * 获取当前机器指纹（供 API 调用）
 */
function getMachineFingerprint() {
  return generateFingerprint();
}

module.exports = { verifyLicense, getMachineFingerprint };
```

## 六、Step 4 — 创建 License 生成工具

**文件**: `tools/generate-license.js`

```javascript
#!/usr/bin/env node
/**
 * License 生成工具 — 仅在开发机上运行，绝不交付给客户
 * 
 * 用法:
 *   node tools/generate-license.js \
 *     --fingerprint=a3f8b2c1d4e5f6a7b8c9d0e1f2a3b4c5 \
 *     --customer=张三 \
 *     --company=测试公司 \
 *     --expire=2026-12-31 \
 *     --plan=test
 * 
 * 参数说明:
 *   --fingerprint  必填，客户机器指纹（首次启动程序时显示）
 *   --customer     必填，客户姓名
 *   --company      选填，公司名称
 *   --expire       选填，过期日期 YYYY-MM-DD（默认 6 个月后）
 *   --plan         选填，授权计划 test/standard/enterprise（默认 test）
 *   --max-devices  选填，最大设备数（默认 1）
 *   --output       选填，输出文件路径（默认 ./license.key）
 */

const crypto = require('crypto');
const fs = require('fs');
const path = require('path');

// ⭐ 密钥必须与 backend/src/license/constants.js 中的 LICENSE_SECRET 完全一致
const LICENSE_SECRET = 'octowork-2026-CHANGE-THIS-TO-YOUR-OWN-SECRET-KEY-64chars-minimum';

// ==================== 参数解析 ====================

function parseArgs() {
  const args = {};
  process.argv.slice(2).forEach(arg => {
    if (arg.startsWith('--')) {
      const [key, ...valueParts] = arg.substring(2).split('=');
      args[key] = valueParts.join('=') || true;
    }
  });
  return args;
}

// ==================== 默认过期时间（6 个月后） ====================

function getDefaultExpire() {
  const d = new Date();
  d.setMonth(d.getMonth() + 6);
  return d.toISOString().split('T')[0];
}

// ==================== 生成 License ====================

function generateLicense(options) {
  const payload = {
    customer: options.customer,
    company: options.company || '',
    fingerprint: options.fingerprint,
    expire: options.expire,
    plan: options.plan || 'test',
    maxDevices: parseInt(options.maxDevices) || 1,
    issuedAt: new Date().toISOString().split('T')[0],
  };

  // HMAC-SHA256 签名（key 排序保证一致性）
  const sortedKeys = Object.keys(payload).sort();
  const payloadStr = JSON.stringify(payload, sortedKeys);
  const signature = crypto
    .createHmac('sha256', LICENSE_SECRET)
    .update(payloadStr)
    .digest('hex');

  payload.signature = signature;

  // Base64 编码
  return Buffer.from(JSON.stringify(payload)).toString('base64');
}

// ==================== 主流程 ====================

const args = parseArgs();

// 校验必填参数
if (!args.fingerprint || !args.customer) {
  console.log('');
  console.log('🔑 OctoWork License 生成工具');
  console.log('');
  console.log('用法:');
  console.log('  node tools/generate-license.js \\');
  console.log('    --fingerprint=a3f8b2c1d4e5f6a7b8c9d0e1f2a3b4c5 \\');
  console.log('    --customer=张三 \\');
  console.log('    --company=测试公司 \\');
  console.log('    --expire=2026-12-31 \\');
  console.log('    --plan=test');
  console.log('');
  console.log('参数:');
  console.log('  --fingerprint  必填  客户机器指纹（首次启动时显示）');
  console.log('  --customer     必填  客户姓名');
  console.log('  --company      选填  公司名称');
  console.log('  --expire       选填  过期日期 YYYY-MM-DD（默认 6 个月后）');
  console.log('  --plan         选填  授权计划 test/standard/enterprise');
  console.log('  --max-devices  选填  最大设备数（默认 1）');
  console.log('  --output       选填  输出路径（默认 ./license.key）');
  console.log('');
  console.log('获取客户指纹: 让客户首次启动程序，屏幕上会显示指纹码');
  console.log('');
  process.exit(1);
}

// 生成
const expire = args.expire || getDefaultExpire();
const outputPath = args.output || path.join(process.cwd(), 'license.key');

const license = generateLicense({
  customer: args.customer,
  company: args.company || '',
  fingerprint: args.fingerprint,
  expire: expire,
  plan: args.plan || 'test',
  maxDevices: args['max-devices'] || 1,
});

// 写入文件
fs.writeFileSync(outputPath, license);

// 输出摘要
console.log('');
console.log('╔═══════════════════════════════════════════════════════════╗');
console.log('║              License 已成功生成                            ║');
console.log('╠═══════════════════════════════════════════════════════════╣');
console.log(`║  客户:     ${args.customer}`);
console.log(`║  公司:     ${args.company || '(未填写)'}`);
console.log(`║  指纹:     ${args.fingerprint}`);
console.log(`║  过期:     ${expire}`);
console.log(`║  计划:     ${args.plan || 'test'}`);
console.log(`║  设备数:   ${args['max-devices'] || 1}`);
console.log(`║  签发日:   ${new Date().toISOString().split('T')[0]}`);
console.log('║');
console.log(`║  输出文件: ${outputPath}`);
console.log('╚═══════════════════════════════════════════════════════════╝');
console.log('');
console.log('📋 下一步操作:');
console.log(`   1. 将 ${path.basename(outputPath)} 发送给客户`);
console.log('   2. 客户将文件放到 ~/octowork-chat/license.key');
console.log('   3. 客户重新启动程序即可');
console.log('');
```

## 十二、安全注意事项

### 12.1 必须保密的内容

| 内容 | 存在位置 | 泄露后果 |
|------|---------|---------|
| `LICENSE_SECRET` 密钥 | `constants.js` + `generate-license.js` | 任何人可自行生成 License |
| `generate-license.js` 工具 | `tools/` 目录 | 同上 |
| 签名算法细节 | `verifier.js` | 可伪造签名 |

### 12.2 安全措施

1. **`tools/` 目录不打包进发布包** — `build-release.sh` 中排除
2. **`constants.js` 编译进字节码** — 用户看不到密钥
3. **`verifier.js` 编译进字节码** — 用户看不到校验逻辑
4. **`license.key` 加入 `.gitignore`** — 不入库
5. **`LICENSE_SECRET` 首次使用前必须更换** — 不要用文档中的示例值

### 12.3 构建时自动排除检查

```bash
# build-release.sh 中加入
if grep -r "CHANGE-THIS" backend/src/license/constants.js; then
  echo "❌ 检测到默认密钥！请先修改 LICENSE_SECRET"
  exit 1
fi
```
